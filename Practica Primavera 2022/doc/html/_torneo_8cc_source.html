86BinTree<Jugador> Torneo::crear_torneo_i(Jugador ant, const vector<Jugador> &rank, int altura, int altura_act, int x)
87{
88 if (altura == altura_act)
89 {
90 BinTree<Jugador> a(ant);
91 return a;
92 }
93 else if (altura - 1 == altura_act and ant.consultar_pos_ranking() <= x)
94 {
95 BinTree<Jugador> a(ant);
96 return a;
97 }
98 else
99 {
100 int jug = pow(2, altura_act) + 1 - ant.consultar_pos_ranking();
101
102 BinTree<Jugador> L = crear_torneo_i(ant, rank, altura, altura_act + 1, x);
103 BinTree<Jugador> R = crear_torneo_i(rank[jug - 1], rank, altura, altura_act + 1, x);
104
105 BinTree<Jugador> Cuadro(ant, L, R);
106
107 return Cuadro;
108 }

139Jugador Torneo::ganador(Jugador &J1, Jugador &J2, string resultados, int pts)
140{
141 int i, j;
142 i = 0;
143 j = 2;
144
145 int sets1, sets2, tot;
146 sets1 = sets2 = tot = 0;
147
148 Jugador Ganador;
149
150 if (resultados.length() > 3)
151 {
152 while (i <= resultados.length())
153 {
154
155 if (resultados[i] - '0' > resultados[j] - '0')
156 {
157 ++sets1;
158 }
159 else
160 {
161 ++sets2;
162 }
163 ++tot;
164 i += 4;
165 j += 4;
166 }
167 if (sets1 > sets2)
168 {
169 J2.modificar_puntos(pts);
170 parts[J2.consultar_pos_ranking() - 1] = J2;
171
172 Ganador = J1;
173 }
174 else
175 {
176 J1.modificar_puntos(pts);
177 parts[J1.consultar_pos_ranking() - 1] = J1;
178
179 Ganador = J2;
180 }
181 }
182 else
183 {
184
185 if (resultados[i] == '1' and resultados[j] == '0')
186 {
187 J2.modificar_puntos(pts);
188 parts[J2.consultar_pos_ranking() - 1] = J2;
189
190 Ganador = J1;
191 }
192 else if (resultados[i] == '0' and resultados[j] == '1')
193 {
194 J1.modificar_puntos(pts);
195 parts[J1.consultar_pos_ranking() - 1] = J1;
196
197 Ganador = J2;
198 }
199 }
200 return Ganador;

202BinTree<Partido> Torneo::leer_resultados_i(const BinTree<Jugador> &A, Jugador &Ganador, const vector<int> &pts, int &nivel, Cjt_Jugadores &General)
203{
204 string puntos;
205 cin >> puntos;
206 if (puntos == "0")
207 {
208 --nivel;
209 Ganador = A.value();
210 BinTree<Partido> a;
211 return a;
212 }
213 else
214 {
215
216 Partido match;
217 match.puntos = puntos;
218 BinTree<Partido> I = leer_resultados_i(A.left(), match.j1, pts, ++nivel, General);
219 BinTree<Partido> D = leer_resultados_i(A.right(), match.j2, pts, ++nivel, General);
220
221 --nivel;
222 Ganador = ganador(match.j1, match.j2, puntos, pts[nivel]);
223
224 General.modificar_estadisticas(Ganador, match.j1, match.j2, puntos, pts, nivel);
225 parts[Ganador.consultar_pos_ranking() - 1] = Ganador;
226
227 BinTree<Partido> res(match, I, D);
228
229 return res;
230 }
